
import java.util.Scanner;
import java.util.Random;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        Random random = new Random();
        int coins = 100;
//        boolean proga = true;

        while (coins > 0) {
            System.out.println("Dice Game!");
            System.out.println("You will play against the robot.");

            System.out.println("Press Enter to roll the dice");
            scanner.nextLine();
            int playerRoll = random.nextInt(6) + 1;
            System.out.println("You rolled a " + playerRoll);


            System.out.println("Robot's turn...");
            int robotRoll = random.nextInt(6) + 1;
            System.out.println("Robot rolled a " + robotRoll);


            if (playerRoll > robotRoll) {
                coins += 10;
                System.out.println("You win!/////////Your money:" + coins);
            } else if (robotRoll > playerRoll) {
                coins -= 25;
                System.out.println("Robot wins!/////////Your money:" + coins);
            } else {
                System.out.println("Draw!");
            }
        }
    }
}
