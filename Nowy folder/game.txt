import java.util.Random;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        Random random = new Random();

        System.out.println("Welcome to the Dice Game!");
        System.out.print("Enter the number of players: ");
        int players = scanner.nextInt();

        int highestRoll = 0;
        int winner = -1;

        for (int i = 1; i <= players; i++) {
            System.out.println("Player " + i + ", press Enter to roll the dice...");
            scanner.nextLine(); // Wait for Enter
            int roll = random.nextInt(6) + 1; // Random number between 1 and 6
            System.out.println("Player " + i + " rolled a " + roll);

            if (roll > highestRoll) {
                highestRoll = roll;
                winner = i;
            }
        }

        System.out.println("The winner is Player " + winner + " with a roll of " + highestRoll + "!");
        scanner.close();
    }
}
